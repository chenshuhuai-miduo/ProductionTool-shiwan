# 数据库设计

## 📋 概述

本文档详细定义了产线采集关联软件的数据库设计，包括表结构设计、==

-- =======================
-- 生产订单表
-- =======================
CREATE TABLE ProductionOrder (
    Id INT AUTO_INCREMENT PRIMARY KEY COMMENT '自增主键',
    SrcId INT NOT NULL COMMENT '原订单表id',
    Type TINYINT NOT NULL COMMENT '类型 (1: 有箱码, 2: 无箱码)',
    OrderNo VARCHAR(255) NULL COMMENT '订单编号',
    OrderSumCoun INT NOT NULL COMMENT '生产总数量',
    ProductSumCount INT NOT NULL COMMENT '订单总数量',
    DmakeDate DATETIME NOT NULL COMMENT '制单日期',
    OrderStatus INT NOT NULL COMMENT '订单状态 (0: 待入库, 1: 入库中, 2: 入库完成, 5: 提前结单)',
    ProductSourceType INT NOT NULL COMMENT '产品来源类型 (0: 默认, 1: PDA, 2: 产线)',
    CreateTime DATETIME NOT NULL COMMENT '创建日期',
    LastModifyTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后修改时间',
    ProductionLine INT NOT NULL COMMENT '所属生产线'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='生产订单表';

-- =======================
-- 生产订单明细表
-- =======================
CREATE TABLE ProductionOrderDetail (
    Id INT AUTO_INCREMENT PRIMARY KEY COMMENT '自增主键',
    OrderCount INT NOT NULL COMMENT '已生产数量',
    ProductCount INT NOT NULL COMMENT '计划订单数量',
    ProductID INT NOT NULL COMMENT '产品ID',
    ProductName VARCHAR(255) NULL COMMENT '产品名称',
    ProductNO VARCHAR(255) NULL COMMENT '产品编号',
    ProductForMatID INT NOT NULL COMMENT '规格ID',
    ProductForMatName VARCHAR(255) NULL COMMENT '规格名称',
    OrderNo VARCHAR(255) NULL COMMENT '订单编号',
    OrderId INT NOT NULL COMMENT '订单ID',
    SyBatchNo VARCHAR(255) NULL COMMENT '批次信息',
    Type INT NOT NULL COMMENT '类型 1 有箱码 2 无箱码',
    Ratio INT NOT NULL COMMENT '拖箱比例',
    IsDel INT NOT NULL COMMENT '是否删除 1删除',
    AddTime DATETIME NOT NULL COMMENT '创建日期',
    ProductTime DATETIME NOT NULL COMMENT '生产时间 (产线设置)',
    TwillendTime DATETIME NOT NULL COMMENT '预计完工时间 (产线设置)',
    CreateTime DATETIME NOT NULL COMMENT '创建日期',
    LastModifyTime TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后修改时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='生产订单明细表';

-- =======================
-- 码关系上传表
-- =======================
CREATE TABLE CodeRelationUpload (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    BiggerSerialNumber VARCHAR(50) NOT NULL COMMENT '跺标',
    BigSerialNumber VARCHAR(50) NOT NULL COMMENT '大标（托盘码）',
    MediumSerialNumber VARCHAR(50) NOT NULL COMMENT '中标',
    SmallSerialNumber VARCHAR(50) NOT NULL COMMENT '小标（箱码）',
    DxCode VARCHAR(255) NOT NULL COMMENT '动销码',
    SalesCode VARCHAR(255) NOT NULL COMMENT '导购码',
    VirtualSerialNumber VARCHAR(50) NOT NULL COMMENT '虚拟跺标',
    IsVirtual INT NOT NULL DEFAULT 0 COMMENT '是否虚拟标签；0否；1是',
    ProductNO VARCHAR(50) NULL COMMENT '产品编号',
    OrderNo VARCHAR(50) NULL COMMENT '订单编号',
    Status INT NOT NULL DEFAULT 0 COMMENT '状态 0未完成 1已完成 2已解除',
    TagNo VARCHAR(255) NOT NULL COMMENT '标签编号',
    Qty INT NOT NULL DEFAULT 0 COMMENT '标签数量（托盘数量）',
    Type INT NOT NULL COMMENT '类型 1有箱码 2无箱码',
    WarehouseNo VARCHAR(50) NULL COMMENT '仓库编号',
    BatchNo VARCHAR(50) NULL COMMENT '批次号',
    AddTime DATETIME NOT NULL COMMENT '添加时间',
    ErrCount INT NOT NULL DEFAULT 0 COMMENT '失败次数',
    Msg VARCHAR(2000) NULL COMMENT '备注',
    UploadTime DATETIME NULL COMMENT '上传时间',
    IsUpload INT NOT NULL DEFAULT 1 COMMENT '是否同步（默认0）0同步 1不同步',
    IsDel INT NOT NULL DEFAULT 0 COMMENT '是否删除',
    TeamName VARCHAR(255) NULL COMMENT '班组名称'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='码关系上传表';

-- =======================
-- 操作日志表
-- =======================
CREATE TABLE OperateLog (
    ID BIGINT AUTO_INCREMENT PRIMARY KEY COMMENT '日志唯一标识',
    OperatorName VARCHAR(50) NOT NULL COMMENT '操作人姓名',
    OperatorIp VARCHAR(50) NULL COMMENT '操作人IP地址',
    ModuleName VARCHAR(100) NOT NULL COMMENT '操作模块',
    Type INT NOT NULL COMMENT '1 操作日志 2 系统日志',
    OperateType VARCHAR(50) NOT NULL COMMENT '操作类型',
    TargetId VARCHAR(50) NULL COMMENT '操作对象ID',
    TargetName VARCHAR(100) NULL COMMENT '操作对象名称',
    OperateContent TEXT NULL COMMENT '操作内容',
    BeforeData TEXT NULL COMMENT '操作前数据(JSON)',
    AfterData TEXT NULL COMMENT '操作后数据(JSON)',
    OperateResult VARCHAR(20) NOT NULL COMMENT '操作结果',
    FailReason TEXT NULL COMMENT '失败原因',
    OperateTime DATETIME(3) NOT NULL DEFAULT CURRENT_TIMESTAMP(3) COMMENT '操作时间',
    DeviceInfo VARCHAR(200) NULL COMMENT '操作设备',
    Remark VARCHAR(500) NULL COMMENT '备注信息'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='操作日志表';

-- 索引
CREATE INDEX idx_operatelog_modulename ON OperateLog(ModuleName);
CREATE INDEX idx_operatelog_operatetime ON OperateLog(OperateTime);
CREATE INDEX idx_operatelog_operatetype ON OperateLog(OperateType);
