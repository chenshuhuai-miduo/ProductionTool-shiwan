--一、码包导入记录表 code_package_import
CREATE TABLE `CodePackageImport` (
    `Id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键',
    `PackageType` TINYINT DEFAULT 1 COMMENT '码包类型：1 / 2 / 3（盖外码小标/盒外码中标/箱外码大标）',
    `PackageName` VARCHAR(200) DEFAULT NULL COMMENT '码包名称（文件名或服务端名称）',
    `ImportTime` DATETIME NOT NULL COMMENT '导入时间',
    `ImportSource` TINYINT DEFAULT 1 COMMENT '导入方式：1 在线 / 2 本地',
    `CodeCount` INT DEFAULT 0 COMMENT '该次导入去重后的码数量',
    `Status` TINYINT DEFAULT 1 COMMENT '状态：1 正常/ -1 删除',
    `Remark` VARCHAR(500) DEFAULT NULL COMMENT '备注',
    `FileName` VARCHAR(255) DEFAULT NULL COMMENT '本地导入文件名（用于重复判断）',
    `CreateTime` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
    `UpdateTime` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    PRIMARY KEY (`Id`),

    INDEX `IdxPackageType` (`PackageType`),
    INDEX `IdxImportTime` (`ImportTime`),
    INDEX `IdxStatus` (`Status`),
    INDEX `IdxImportSource` (`ImportSource`)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='码包导入记录表';

--二、码包码热表 code_package_item_hot
--用于 未被采集的码（热数据）
CREATE TABLE `CodePackageItemHot` (
    `Id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键',
    `ImportId` BIGINT NOT NULL COMMENT '关联 CodePackageImport.Id',
    `PackageType` TINYINT DEFAULT 1 COMMENT '码包类型：1 / 2 / 3（盖外码小标/盒外码中标/箱外码大标）',
    `CodeValue` VARCHAR(64) NOT NULL COMMENT '码值（瓶/盒/箱外码）',
    `CreateTime` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

    PRIMARY KEY (`Id`),

    UNIQUE KEY `UkPackageCode` (`PackageType`,`CodeValue`),

    -- 说明：大批量本地导入时，与库内已存在码的比对通过视图 AllCodePackageCodes 及（可选）临时表 JOIN 完成；
    -- UkPackageCode 保证同类型码不重复入库，IdxCodeValue 辅助按码值检索。

    INDEX `IdxImportId` (`ImportId`),
    INDEX `IdxPackageType` (`PackageType`),
    INDEX `IdxCodeValue` (`CodeValue`)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='码包码热表（未被采集的码）';

--三、码包码冷表 code_package_item_cold
--用于 已经被采集关联的码（冷数据）
CREATE TABLE `CodePackageItemCold` (
    `Id` BIGINT NOT NULL AUTO_INCREMENT COMMENT '主键',
    `ImportId` BIGINT NOT NULL COMMENT '关联 CodePackageImport.Id',
    `PackageType` TINYINT DEFAULT 1 COMMENT '码包类型：1 / 2 / 3（盖外码小标/盒外码中标/箱外码大标）',
    `CodeValue` VARCHAR(64) NOT NULL COMMENT '码值',
    `AssociatedAt` DATETIME DEFAULT NULL COMMENT '该码首次被采集/关联时间',
    `CreateTime` DATETIME DEFAULT CURRENT_TIMESTAMP COMMENT '落冷时间',
    `UpdateTime` DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

    PRIMARY KEY (`Id`),

    INDEX `IdxImportId` (`ImportId`),
    INDEX `IdxPackageType` (`PackageType`),
    INDEX `IdxCodeValue` (`CodeValue`),
    INDEX `IdxImportCode` (`ImportId`,`CodeValue`)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='码包码冷表（已被采集的码）';

--四、码包全码视图 all_code_package_codes
CREATE VIEW `AllCodePackageCodes` AS
SELECT `PackageType`, `CodeValue` FROM `CodePackageItemHot`
UNION ALL
SELECT `PackageType`, `CodeValue` FROM `CodePackageItemCold`;

--五、在线拉取 start_time 规则
--不再使用 `CodePackagePullTime` 表。在线更新时按码包类型（小标=1、大标=3）查询
--`CodePackageImport` 中 `ImportSource = 1` 且 `Status = 1` 的最近 `CreateTime`，取两者较早值作为
--开放平台 `start_time`；若某类型无在线导入记录，则回退到系统默认起始时间。

--六、 CodeRelationUpload（码关系上传表）
DROP TABLE IF EXISTS CodeRelationUpload;
CREATE TABLE CodeRelationUpload (
  ID INT NOT NULL AUTO_INCREMENT COMMENT '主键ID',
  BiggerSerialNumber VARCHAR(50) NOT NULL DEFAULT '' COMMENT '跺标(垛码，即虚拟垛标)',
  BigSerialNumber VARCHAR(50) NOT NULL DEFAULT '' COMMENT '大标（箱码）',
  MediumSerialNumber VARCHAR(50) NOT NULL DEFAULT '' COMMENT '中标（盒码）',
  SmallSerialNumber VARCHAR(50) NOT NULL DEFAULT '' COMMENT '小标（瓶码）',
  DxCode VARCHAR(255) NOT NULL DEFAULT '' COMMENT '动销码',
  SalesCode VARCHAR(255) NOT NULL DEFAULT '' COMMENT '导购码',
  VirtualSerialNumber VARCHAR(50) NOT NULL DEFAULT '' COMMENT '虚拟跺标',
  IsVirtual INT NOT NULL DEFAULT 0 COMMENT '是否虚拟标签：0 否；1 是',
  ProductNO VARCHAR(50) NOT NULL DEFAULT '' COMMENT '产品编号',
  OrderNo VARCHAR(50) NOT NULL DEFAULT '' COMMENT '订单编号（M1同步及手工采集时为空串，盒箱关联后回填）',
  Status INT NOT NULL DEFAULT 0 COMMENT '状态：0 未完成；1 已完成；2 已解除；3 未成垛 4 待剔除',
  TagNo VARCHAR(255) NOT NULL DEFAULT '' COMMENT '标签编号',
  Qty INT NOT NULL DEFAULT 0 COMMENT '标签数量（托盘数量）',
  Type INT NOT NULL COMMENT '类型：1 有箱码；2 无箱码',
  WarehouseNo VARCHAR(50) NOT NULL DEFAULT '' COMMENT '仓库编号',
  BatchNo VARCHAR(50) NOT NULL DEFAULT '' COMMENT '批次号',
  AddTime DATETIME NOT NULL COMMENT '创建时间',
  PalletTime DATETIME DEFAULT NULL COMMENT '成垛时间（仅生成虚拟垛标时写入）',
  ErrCount INT NOT NULL DEFAULT 0 COMMENT '失败次数',
  Msg VARCHAR(2000) DEFAULT NULL COMMENT '备注信息',
  UploadTime DATETIME DEFAULT NULL COMMENT '上传时间',
  IsUpload INT NOT NULL DEFAULT 0 COMMENT '是否上传：0 未上传；1 已上传 2 上传失败 3 上传中',
  IsDel INT NOT NULL DEFAULT 0 COMMENT '是否删除：0 未删除；1 删除',
  TeamName VARCHAR(255) NOT NULL DEFAULT '' COMMENT '班组名称',
  LastModifyTime TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '最后修改时间',
  PRIMARY KEY (ID),

  -- ① 箱码精确查重/按箱码找订单/取消关联CASE操作
  --    isCaseCodeAlreadyUsed: WHERE BigSerialNumber=? AND IsDel=0
  --    getTaskByCaseCode:     WHERE BigSerialNumber=? AND IsDel=0 LIMIT 1
  --    checkCancelable CASE:  countActiveBy("BigSerialNumber")
  KEY idx_bigserial_del         (BigSerialNumber, IsDel),

  -- ② 盒码三种校验条件共走此索引（BigSerialNumber 作第3列支持=''与!=''区间）
  --    validateBoxCodeForReceive: WHERE MediumSerialNumber=? AND IsDel=0 AND BigSerialNumber=''
  --    isBoxCodeAlreadyUsed:      WHERE MediumSerialNumber=? AND IsDel=0 AND BigSerialNumber!=''
  --    validateManualCode:         WHERE MediumSerialNumber=? AND IsDel=0
  --    associateCaseCodeWithBoxCodes: WHERE IsDel=0 AND MediumSerialNumber IN(...)
  --    getProblematicBoxCodes:    WHERE MediumSerialNumber IN(...) AND Status IN(4,5,6) AND IsDel=0
  KEY idx_medium_del_big        (MediumSerialNumber, IsDel, BigSerialNumber),

  -- ③ 瓶码防重/手工采集校验/M1同步防重/取消关联BOTTLE
  --    validateManualCode: WHERE SmallSerialNumber=? AND IsDel=0
  --    smallCodeExists:    WHERE IsDel=0 AND SmallSerialNumber=?
  KEY idx_small_del             (SmallSerialNumber, IsDel),

  -- ④ 订单维度核心索引（成垛、计数、TagNo恢复、产品同步等所有订单维度操作）
  --    countCurrentCasesInPallet: WHERE OrderNo=? AND IsDel=0 AND VirtualSerialNumber='' AND BigSerialNumber!=''
  --    completeCurrentPallet:     WHERE OrderNo=? AND IsDel=0 AND VirtualSerialNumber='' AND BigSerialNumber!=''
  --    getProducedPalletCount:    WHERE OrderNo=? AND IsDel=0 AND VirtualSerialNumber!=''
  --    getOrCreateCurrentTagNo:   WHERE OrderNo=? AND IsDel=0 AND VirtualSerialNumber=''
  --    syncUnpalletProductNo:     WHERE IsDel=0 AND OrderNo=? AND VirtualSerialNumber=''
  --    existsOrderNo:             WHERE OrderNo=? AND IsDel=0
  --    associateCaseCodeWithPendingBoxes: WHERE IsDel=0 AND (OrderNo='' OR OrderNo=?) AND BigSerialNumber=''
  KEY idx_order_del_virtual_big (OrderNo(50), IsDel, VirtualSerialNumber(50), BigSerialNumber(50)),

  -- ⑤ 【关键】垛码精确查找（原缺失，千万级下多处高频操作无索引）
  --    isPalletUploaded:         WHERE VirtualSerialNumber=? AND IsDel=0
  --    queryPalletStatus:        WHERE VirtualSerialNumber=? AND IsDel=0
  --    syncVirtualPalletToCloud: WHERE VirtualSerialNumber=? AND IsDel=0
  --    UPDATE IsUpload=3/1/2:    WHERE VirtualSerialNumber=? AND IsDel=0
  --    doLocalCancel PALLET:     WHERE VirtualSerialNumber=? AND IsDel=0
  --    checkCancelable PALLET:   countActiveBy("VirtualSerialNumber") + COUNT DISTINCT BigSerialNumber
  KEY idx_virtualserial_del     (VirtualSerialNumber(50), IsDel),

  -- ⑥ 待上传垛扫描（修正原 idx_isupload_bigger 错误列名，改为 VirtualSerialNumber）
  --    getPendingUploadPallets: WHERE IsUpload=0 AND VirtualSerialNumber!='' AND IsDel=0
  --    getUploadRecords:        WHERE UploadTime BETWEEN ? AND ? AND VirtualSerialNumber!='' AND IsUpload IN(1,2)
  KEY idx_isupload_virtual_del  (IsUpload, VirtualSerialNumber(50), IsDel),

  -- ⑦ 暂存恢复/剔除状态批量更新（修正原 idx_status_isupload，补充 IsDel）
  --    getUnfinishedOrders: WHERE Status=3 AND IsDel=0 AND BigSerialNumber!=''
  --    checkAndWriteRejectsForProblematic UPDATE: WHERE MediumSN=? AND SmallSN=? AND IsDel=0 AND Status=0
  KEY idx_status_del            (Status, IsDel),

  -- ⑧ 生产统计/垛列表时间范围查询（合并后单次扫描）
  --    getProductionSummary: WHERE PalletTime BETWEEN ? AND ?
  --    getPalletList:        WHERE PalletTime BETWEEN ? AND ? AND VirtualSerialNumber!=''
  KEY idx_pallettime            (PalletTime),

  -- ⑨ 上传记录时间范围查询
  --    getUploadRecords: WHERE UploadTime BETWEEN ? AND ? AND VirtualSerialNumber!=''
  KEY idx_uploadtime            (UploadTime),

  -- ⑩ 采集时间/待上传垛 ORDER BY MIN(AddTime) 排序辅助
  --    getPendingUploadPallets: ORDER BY MIN(AddTime) ASC
  --    CodePackageRelationMapper: SELECT AddTime WHERE xSN IN(...)
  KEY idx_addtime               (AddTime)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='码关系上传表（存储箱码、托盘码、垛码等多级码关系）';
